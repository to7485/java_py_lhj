SELECT 
테이블에서 데이터를 조회하기 위해 사용하는 키워드

SELECT 사원번호, 이름, 나이, 성별, 급여, 입사날짜 FROM 테이블;

전체속성에 대한 데이터를 보고싶다.
SELECT * FROM 테이블;

조건을 줘서 내가 원하는 데이터만 조회하겠다.
WHERE 절을 추가해야 한다.
- FORM 뒤에 작성을 한다.
- WHERE 조건식

속성에 대해서 비교를 시행한다.

급여(SALARY) = 100000
입사날짜 > '1980-01-01'

조건 2개 이상을 비교하려 할 때 
AND -> BETWEEN
 
OR -> IN

글자를 포함하냐?
WHERE 컬럼명 LIKE '조건';

% : 0글자 이상

_ : 한글자 취급

A% : A로 시작하는 데이터(A뒤로 몇글자여도 상관X)

A_ : A로 시작하는 두글자 데이터

INSERT
데이터를 테이블에 추가하는 키워드

INSERT INTO 테이블명 (컬럼1,컬럼2,...)
VALUES(값1,값2,...)

INSERT INTO 테이블명 VALUES(값1,값2,...)

CREATE TABLE USER(
	이름 VRCHAR(100) PRIMARY KEY,
	나이 INT,
	주소 VARCHAR(300) DEFAULT '집없음'
)

INSERT INTO USER (이름, 나이, 주소)
VALUES('홍길동',25,'인천시 부평구');

컬럼에 DEFAULT 제약조건이 지정이 되어 있을 때 
INSERT할 때 컬럼을 생략하면 된다.

INSERT INTO USER(이름,나이)
VALUES('박길동',20);

DEFAULT 키워드 사용하기

INSERT INTO USER(이름,나이,주소)
VALUES('박길동',20,DEFAULT);

컬럼목록을 생략해도 된다.
컬럼목록을 생략해도 DEFAULT 키워드를 사용하면 DEFAULT를 사용할 수 있다.
INSERT INTO USER VALUES('박길동',20,DEFAULT);
